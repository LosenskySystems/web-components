import { Slider, Label } from '@losensky-systems/web-components-core'

# Slider Component

The Slider component allows users to select a numeric value or range of values by dragging thumbs along a track. It supports both single value and dual-thumb range modes.

## Import

```tsx
import { Slider } from '@losensky-systems/web-components-core'
```

## Basic Usage

<CodeBlock code={`<Slider />`}>
  <div className="mb-8">
    <Slider />
  </div>
</CodeBlock>

## Range Mode

<CodeBlock code={`<Slider range defaultValue={[20, 80]} />`}>
  <div className="mb-8">
    <Slider range defaultValue={[20, 80]} />
  </div>
</CodeBlock>

## Controlled Usage

<CodeBlock code={`const [value, setValue] = useState(50);
const [rangeValue, setRangeValue] = useState([20, 80]);

<Slider value={value} onChange={setValue} />
<Slider range value={rangeValue} onChange={setRangeValue} />`}>
  <div className="mb-8">
    <div className="mb-4">
      <Label>Controlled Single Value</Label>
      <Slider value={50} onChange={(value) => console.log('Single value:', value)} />
    </div>
    
    <div className="mb-4">
      <Label>Controlled Range Value</Label>
      <Slider range value={[20, 80]} onChange={(value) => console.log('Range value:', value)} />
    </div>
  </div>
</CodeBlock>
