import { Label } from '@losensky-systems/web-components-core'

# Label Component

The Label component provides accessible labels for form elements with proper contrast and styling.

## Import

```tsx
import { Label } from '@losensky-systems/web-components-core'
```

## Basic Usage

<CodeBlock code={`<Label htmlFor="email">Email Address</Label>
<Label htmlFor="password">Password</Label>
<Label htmlFor="name" required>Full Name</Label>`}>
  <div className="mb-8">
    <Label htmlFor="email">Email Address</Label>
    <Label htmlFor="password">Password</Label>
    <Label htmlFor="name" required>Full Name</Label>
  </div>
</CodeBlock>

## Sizes

<CodeBlock code={`<Label size="sm" htmlFor="small">Small Label</Label>
<Label size="md" htmlFor="medium">Medium Label (Default)</Label>
<Label size="lg" htmlFor="large">Large Label</Label>`}>
  <div className="mb-8">
    <Label size="sm" htmlFor="small">Small Label</Label>
    <Label size="md" htmlFor="medium">Medium Label (Default)</Label>
    <Label size="lg" htmlFor="large">Large Label</Label>
  </div>
</CodeBlock>
